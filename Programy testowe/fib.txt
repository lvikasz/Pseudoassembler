TAB DC 12*INTEGER(1)
JEDEN DC INTEGER(1)
ZERO DC INTEGER(0)
CZTERY DC INTEGER(4)
DZIESIEC DC INTEGER(11)

LA 1, TAB
L 0, ZERO

L 4, JEDEN
L 5, JEDEN

ST 4, 0(1)
A 1, CZTERY 
ST 5, 0(1)
A 1, CZTERY  

L 2, JEDEN
SR 3, 3

FIB AR 3, 4
AR 3, 5
ST 3, 0(1)
A 1, CZTERY
LR 5, 4
LR 4, 3
SR 3, 3
A 2, JEDEN
C 2, DZIESIEC
JN FIB

PRT TAB
